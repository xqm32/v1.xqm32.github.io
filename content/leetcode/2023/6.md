```python
# @lc code=start
from typing import List


class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        res = []
        n = len(nums)

        def helper(i, tmp):
            res.append(tmp)
            for j in range(i, n):
                helper(j + 1, tmp + [nums[j]])

        helper(0, [])
        return res
```

```python
#

# @lc code=start
from typing import List


class Solution:
    def minimumCost(self, cost: List[int]) -> int:
        return sum(cost) - sum(sorted(cost, reverse=True)[2::3])
```

```python
# @lc code=start
from typing import List


class Solution:
    def sortPeople(self, names: List[str], heights: List[int]) -> List[str]:
        n = len(names)
        ans = [None] * n
        for i in range(n):
            ans[i] = (names[i], heights[i])
        ans.sort(key=lambda x: (-x[1], x[0]))
        return [x[0] for x in ans]
```

```python
# @lc code=start
from typing import List


class Solution:
    def imageSmoother(self, img: List[List[int]]) -> List[List[int]]:
        m, n = len(img), len(img[0])
        res = [[0] * n for _ in range(m)]

        for i in range(m):
            for j in range(n):
                res[i][j] = self.get_avg(img, i, j)

        return res

    def get_avg(self, img, i, j):
        m, n = len(img), len(img[0])
        sum, count = 0, 0

        for x in range(i - 1, i + 2):
            for y in range(j - 1, j + 2):
                if 0 <= x < m and 0 <= y < n:
                    sum += img[x][y]
                    count += 1

        return sum // count
```

```python
# @lc code=start
class Solution:
    def minLength(self, s: str) -> int:
        while s.find("AB") != -1 or s.find("CD") != -1:
            s = s.replace("AB", "")
            s = s.replace("CD", "")
        return len(s)
```

```python
# @lc code=start
from typing import List


class Solution:
    def findLHS(self, nums: List[int]) -> int:
        nums.sort()
        res = 0
        i = 0
        j = 1
        while j < len(nums):
            if nums[j] - nums[i] == 1:
                res = max(res, j - i + 1)
                j += 1
            elif nums[j] - nums[i] > 1:
                i += 1
            else:
                j += 1
        return res
```

```python
#

# @lc code=start
from typing import List


class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        d = {p[0]: p for p in pieces}
        res = []
        for num in arr:
            res += d.get(num, [])
        return res == arr


# @lc code=end
```

```python
# [1337] 矩阵中战斗力最弱的 K 行
#

# @lc code=start
from typing import List


class Solution:
    def kWeakestRows(self, mat: List[List[int]], k: int) -> List[int]:
        m = [i.count(1) * 1000 + c for c, i in enumerate(mat)]
        return [i % 1000 for i in sorted(m)[:k]]
```
