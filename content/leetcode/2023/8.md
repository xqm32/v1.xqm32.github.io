# [2500] 删除每行中的最大值
#

# @lc code=start
from typing import List


class Solution:
    def deleteGreatestValue(self, grid: List[List[int]]) -> int:
        return sum(max(i) for i in zip(*[sorted(j) for j in grid]))


# @lc code=end
