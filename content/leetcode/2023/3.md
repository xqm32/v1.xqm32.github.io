# [2006] 差的绝对值为 K 的数对数目
#

# @lc code=start
from typing import List


class Solution:
    def countKDifference(self, nums: List[int], k: int) -> int:
        ans = 0
        nums.sort()

        for i in range(len(nums)):
            for j in range(i + 1, len(nums)):
                if abs(nums[i] - nums[j]) == k:
                    ans += 1

        return ans


# @lc code=end
