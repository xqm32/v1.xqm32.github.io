```python
# [2496] 数组中字符串的最大值
#

# @lc code=start
from typing import List


class Solution:
    def maximumValue(self, strs: List[str]) -> int:
        def value(s: str):
            if s.isdigit():
                return int(s)
            else:
                return len(s)

        return max(map(value, strs))
```

```python
# [2068] 检查两个字符串是否几乎相等
#

# @lc code=start
from collections import Counter


class Solution:
    def checkAlmostEquivalent(self, word1: str, word2: str) -> bool:
        c1 = Counter(word1)
        c2 = Counter(word2)
        keys = set(c1.keys()).union(set(c2.keys()))
        for i in keys:
            if i not in c1 and c2[i] > 3:
                return False
            elif i not in c2 and c1[i] > 3:
                return False
            else:
                if abs(c2[i] - c1[i]) > 3:
                    return False
        return True
```

```python
# @lc app=leetcode.cn id=2016 lang=python3
#
# [2016] 增量元素之间的最大差值
#

# @lc code=start
from typing import List


class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        min_num = nums[0]
        max_diff = -1
        for i in range(1, n):
            if nums[i] > min_num:
                max_diff = max(max_diff, nums[i] - min_num)
            else:
                min_num = nums[i]
        return max_diff


# @lc code=end
```

```python
# [2032] 至少在两个数组中出现的值
#

# @lc code=start
from typing import List


class Solution:
    def twoOutOfThree(
        self, nums1: List[int], nums2: List[int], nums3: List[int]
    ) -> List[int]:
        return list(
            set(nums1) & set(nums2) | set(nums1) & set(nums3) | set(nums2) & set(nums3)
        )
```

```python
# [2520] 统计能整除数字的位数
#


# @lc code=start
class Solution:
    def countDigits(self, num: int) -> int:
        ans = 0
        for i in str(num):
            if num % int(i) == 0:
                ans += 1
        return ans
```
