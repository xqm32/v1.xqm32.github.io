#

# @lc code=start
from typing import List


class Solution:
    def distanceBetweenBusStops(
        self, distance: List[int], start: int, destination: int
    ) -> int:
        if start > destination:
            start, destination = destination, start
        return min(
            sum(distance[start:destination]),
            sum(distance[destination:] + distance[:start]),
        )

# [733] 图像渲染
#

# @lc code=start
from typing import List


class Solution:
    def floodFill(
        self, image: List[List[int]], sr: int, sc: int, color: int
    ) -> List[List[int]]:
        def dfs(image, sr, sc, color, newColor):
            if image[sr][sc] == color:
                image[sr][sc] = newColor
                if sr >= 1:
                    dfs(image, sr - 1, sc, color, newColor)
                if sr + 1 < len(image):
                    dfs(image, sr + 1, sc, color, newColor)
                if sc >= 1:
                    dfs(image, sr, sc - 1, color, newColor)
                if sc + 1 < len(image[0]):
                    dfs(image, sr, sc + 1, color, newColor)

        if image[sr][sc] != color:
            dfs(image, sr, sc, image[sr][sc], color)
        return image

class Solution:
    def maxDistance(self, colors: List[int]) -> int:
        n = len(colors)
        res = 0
        for i in range(n):
            for j in range(i + 1, n):
                if colors[i] != colors[j]:
                    res = max(res, j - i)

        if len(s) == 0:
            return []

        return [s[:k]] + self.divideString(s[k:], k, fill)

from typing import List


class Solution:
    def findLengthOfLCIS(self, nums: List[int]) -> int:
        if not nums:
            return 0

        l, t = 1, 1
        for i in range(1, len(nums)):
            if nums[i] - nums[i - 1] > 0:
                t += 1
            else:
                l = max(l, t)
                t = 1

        return max(l, t)

#

# @lc code=start
from typing import List


class Solution:
    def similarPairs(self, words: List[str]) -> int:
        w = []
        for i in words:
            s = sorted(set(i))
            t = "".join(s)
            w.append(t)
        ans = 0
        for i in range(len(w)):
            for j in range(i + 1, len(w)):
                if w[i] == w[j]:
                    ans += 1
        return ans

        ans = 0

#


# @lc code=start
class Solution:
    def reorderSpaces(self, text: str) -> str:
        spaces = text.count(" ")
        words = text.split()
        if len(words) == 1:
            return words[0] + " " * spaces
        each = spaces // (len(words) - 1)
        left = spaces % (len(words) - 1)
        return (" " * each).join(words) + " " * left

class Solution:
    def commonFactors(self, a: int, b: int) -> int:
        if a == 0 or b == 0:
            return 0
        if a > b:
            a, b = b, a
        res = 0
        for i in range(1, a + 1):
            if a % i == 0 and b % i == 0:
                res += 1
        return res

#

# @lc code=start
from collections import Counter


class Solution:
    def checkAlmostEquivalent(self, word1: str, word2: str) -> bool:
        c1 = Counter(word1)
        c2 = Counter(word2)
        keys = set(c1.keys()).union(set(c2.keys()))
        for i in keys:
            if i not in c1 and c2[i] > 3:
                return False
            elif i not in c2 and c1[i] > 3:
                return False
            else:
                if abs(c2[i] - c1[i]) > 3:
                    return False
        return True

#

# @lc code=start
from typing import List


class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        min_num = nums[0]
        max_diff = -1
        for i in range(1, n):
            if nums[i] > min_num:
                max_diff = max(max_diff, nums[i] - min_num)
            else:
                min_num = nums[i]
        return max_diff
