```python
# [2032] 至少在两个数组中出现的值
#

# @lc code=start
from typing import List


class Solution:
    def twoOutOfThree(
        self, nums1: List[int], nums2: List[int], nums3: List[int]
    ) -> List[int]:
        return list(
            set(nums1) & set(nums2) | set(nums1) & set(nums3) | set(nums2) & set(nums3)
        )
```

```python
#
# [2520] 统计能整除数字的位数
#


# @lc code=start
class Solution:
    def countDigits(self, num: int) -> int:
        ans = 0
        for i in str(num):
            if num % int(i) == 0:
                ans += 1
        return ans


# @lc code=end
```
