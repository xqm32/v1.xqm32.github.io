```python
import re


class Solution:
    def areNumbersAscending(self, s: str) -> bool:
        a = 0
        for i in re.findall(r"\d+", s):
            if a >= int(i):
                return False
            a = int(i)
        return True
```

```python
# [2427] 公因子的数目
#


# @lc code=start
class Solution:
    def commonFactors(self, a: int, b: int) -> int:
        if a == 0 or b == 0:
            return 0
        if a > b:
            a, b = b, a
        res = 0
        for i in range(1, a + 1):
            if a % i == 0 and b % i == 0:
                res += 1
        return res
```

```python
# [2566] 替换一个数字后的最大差值
#


# @lc code=start
class Solution:
    def minMaxDifference(self, num: int) -> int:
        str_num = str(num)
        max_num = num
        min_num = num

        for i in range(10):
            str_num_9 = str_num.replace(str(i), "9")
            num_9 = int(str_num_9)
            if num_9 > max_num:
                max_num = num_9

            str_num_0 = str_num.replace(str(i), "0")
            num_0 = int(str_num_0)
            if num_0 < min_num:
                min_num = num_0

        return max_num - min_num
```

```python
from typing import List


class Solution:
    def maximumValue(self, strs: List[str]) -> int:
        def value(s: str):
            if s.isdigit():
                return int(s)
            else:
                return len(s)

        return max(map(value, strs))
```

```python
# [2068] 检查两个字符串是否几乎相等
#

# @lc code=start
from collections import Counter


class Solution:
    def checkAlmostEquivalent(self, word1: str, word2: str) -> bool:
        c1 = Counter(word1)
        c2 = Counter(word2)
        keys = set(c1.keys()).union(set(c2.keys()))
        for i in keys:
            if i not in c1 and c2[i] > 3:
                return False
            elif i not in c2 and c1[i] > 3:
                return False
            else:
                if abs(c2[i] - c1[i]) > 3:
                    return False
        return True
```

```python
# @lc app=leetcode.cn id=2016 lang=python3
#
# [2016] 增量元素之间的最大差值
#

# @lc code=start
from typing import List


class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        min_num = nums[0]
        max_diff = -1
        for i in range(1, n):
            if nums[i] > min_num:
                max_diff = max(max_diff, nums[i] - min_num)
            else:
                min_num = nums[i]
        return max_diff


# @lc code=end
```

```python
class Solution:
    def twoOutOfThree(
        self, nums1: List[int], nums2: List[int], nums3: List[int]
    ) -> List[int]:
        return list(
            set(nums1) & set(nums2) | set(nums1) & set(nums3) | set(nums2) & set(nums3)
        )
```
