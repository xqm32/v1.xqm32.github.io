```python
# [2016] 增量元素之间的最大差值
#

# @lc code=start
from typing import List


class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        n = len(nums)
        min_num = nums[0]
        max_diff = -1
        for i in range(1, n):
            if nums[i] > min_num:
                max_diff = max(max_diff, nums[i] - min_num)
            else:
                min_num = nums[i]
        return max_diff
```

```python

# @lc app=leetcode.cn id=2032 lang=python3
#
# [2032] 至少在两个数组中出现的值
#

# @lc code=start
from typing import List


class Solution:
    def twoOutOfThree(
        self, nums1: List[int], nums2: List[int], nums3: List[int]
    ) -> List[int]:
        return list(
            set(nums1) & set(nums2) | set(nums1) & set(nums3) | set(nums2) & set(nums3)
        )


# @lc code=end
```

```python
#
# [2520] 统计能整除数字的位数
#


# @lc code=start
class Solution:
    def countDigits(self, num: int) -> int:
        ans = 0
        for i in str(num):
            if num % int(i) == 0:
                ans += 1
        return ans


# @lc code=end
```
