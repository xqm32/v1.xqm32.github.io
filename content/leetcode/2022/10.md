```python
#
# @lc app=leetcode.cn id=1009 lang=python3
#
# [1009] 十进制整数的反码
#

# @lc code=start
class Solution:
    def bitwiseComplement(self, n: int) -> int:
        return int(
            "".join("0" if i == "1" else "1" for i in bin(n).removeprefix("0b")), 2
        )


# @lc code=end
sol = Solution()
for i in [5, 7, 10]:
    print(sol.bitwiseComplement(i))
```
