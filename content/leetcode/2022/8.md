from typing import List


class Solution:
    def diStringMatch(self, s: str) -> List[int]:
        return [
            len(s) - s[:i].count("D") if s[i] == "D" else s[:i].count("I")
            for i in range(len(s))
        ] + [s.count("I")]

        r = []
        i = s.count("I")
        d = s.count("D")

        r.append(i)

        for c in reversed(s):
            if c == "D":
                d -= 1
                r.append(len(s) - d)
            else:
                i -= 1
                r.append(i)

        return list(reversed(r))


# @lc code=end

sol = Solution()
print(sol.diStringMatch("IDID"))
print(sol.diStringMatch("III"))
print(sol.diStringMatch("DDI"))

# @lc code=start
class Solution:
    def makeGood(self, s: str) -> str:
        def nextGood(s: str) -> tuple[str, bool]:
            re = ""
            isGood = True
            i = 0
            while i < len(s) - 1:
                if abs(ord(s[i]) - ord(s[i + 1])) == 32:
                    i += 2
                    isGood = False
                    continue
                re += s[i]
                i += 1
            if i < len(s):
                re += s[i]
            return re, isGood

        isGood = False
        while not isGood:
            s, isGood = nextGood(s)
        return s


# @lc code=end
sol = Solution()

print(sol.makeGood("abBAcC"))
print(sol.makeGood("s"))
print(sol.makeGood(""))

class Solution:
    def countNegatives(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        ans = 0

        j = n - 1
        for i in range(m):
            for j in reversed(range(0, j + 1)):
                if grid[i][j] >= 0:
                    ans += n - j - 1
                    print(f"{ans} += {n} - {j+1}")
                    break
            else:
                ans += n
                print(f"{ans} += {n}")

        return ans


# @lc code=end

from collections import Counter
from typing import List


class Solution:
    def countElements(self, nums: List[int]) -> int:
        a = min(nums)
        b = max(nums)
        if a == b:
            return 0
        d = Counter(nums)
        if d[a] + d[b] == len(nums):
            return 0
        return len(nums) - d[a] - d[b]

from collections import Counter
from functools import reduce
from typing import List


class Solution:
    def hasGroupsSizeX(self, deck: List[int]) -> bool:
        def gcd(a, b):
            return a if b == 0 else gcd(b, a % b)

        deck = Counter(deck).values()
        r = reduce(gcd, deck)
        return r >= 2

from typing import List


class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        if len(prices) < 2:
            return 0

        a = [prices[-1]]
        for i in reversed(prices[:-1]):
            if i > a[-1]:
                a.append(i)
            else:
                a.append(a[-1])
        a = list(reversed(a))

        return max(a[i] - prices[i] for i in range(len(a)))

class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:
        di = dict()
        dj = dict()
        s = s.split()
        if len(s) != len(pattern):
            return False
        for i, j in zip(pattern, s):
            if j in dj and dj[j] != i:
                return False
            elif i in di and di[i] != j:
                return False
            else:
                di[i] = j
                dj[j] = i
        return True
# @lc code=end

sol = Solution()

from typing import Optional


class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


# @lc code=start
# Definition for a binary tree node.


class Solution:
    def sumOfLeftLeaves(self, root: Optional[TreeNode]) -> int:
        def func(root: Optional[TreeNode], isLeft: bool) -> int:
            if root is None:
                return 0
            elif root.left is None and root.right is None and isLeft:
                return root.val
            else:
                return func(root.left, True) + func(root.right, False)

        return func(root, False)

from typing import List


class Solution:
    def isMonotonic(self, nums: List[int]) -> bool:
        for i in range(1, len(nums)):
            if nums[i] < nums[i - 1]:
                nums[i - 1] = -1
            elif nums[i] == nums[i - 1]:
                nums[i - 1] = 0
            elif nums[i] > nums[i - 1]:
                nums[i - 1] = 1
        nums[-1] = 0
        return nums.count(1) == 0 or nums.count(-1) == 0


# @lc code=end
sol = Solution()

from typing import List


class Solution:
    def islandPerimeter(self, grid: List[List[int]]) -> int:
        h = len(grid)
        l = len(grid[0])
        c = 0

        for i in range(h):
            for j in range(l):
                if grid[i][j] == 0:
                    continue
                if i == 0 or grid[i - 1][j] == 0:
                    c += 1
                if i == h - 1 or grid[i + 1][j] == 0:
                    c += 1
                if j == 0 or grid[i][j - 1] == 0:
                    c += 1
                if j == l - 1 or grid[i][j + 1] == 0:
                    c += 1
        return c


# @lc code=end
sol = Solution()

class Solution:
    def isHappy(self, n: int) -> bool:
        def iterNext(n: int) -> int:
            m = 0
            for i in str(n):
                m += int(i) ** 2
            # print(f"{n} -> {m}")

        a = set()
        while (n := iterNext(n)) != 1:
            if n in a:
                return False
            a.add(n)
