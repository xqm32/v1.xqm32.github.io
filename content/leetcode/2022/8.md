```python
# @lc code=start
from typing import List


class Solution:
    def islandPerimeter(self, grid: List[List[int]]) -> int:
        h = len(grid)
        l = len(grid[0])
        c = 0

        for i in range(h):
            for j in range(l):
                if grid[i][j] == 0:
                    continue
                if i == 0 or grid[i - 1][j] == 0:
                    c += 1
                if i == h - 1 or grid[i + 1][j] == 0:
                    c += 1
                if j == 0 or grid[i][j - 1] == 0:
                    c += 1
                if j == l - 1 or grid[i][j + 1] == 0:
                    c += 1
        return c


# @lc code=end
sol = Solution()
print(sol.islandPerimeter([[0, 1, 0, 0], [1, 1, 1, 0], [0, 1, 0, 0], [1, 1, 0, 0]]))
print(sol.islandPerimeter([[1]]))
print(sol.islandPerimeter([[1, 0]]))
```

```python
# [448] 找到所有数组中消失的数字
#

# @lc code=start
from typing import List


class Solution:
    def findDisappearedNumbers(self, nums: List[int]) -> List[int]:
        return sorted(list(set(range(1, len(nums) + 1)) - set(nums)))


# @lc code=end

sol = Solution()
print(sol.findDisappearedNumbers([4, 3, 2, 7, 8, 2, 3, 1]))
print(sol.findDisappearedNumbers([1, 1]))
```

```python
# @lc code=start
class Solution:
    def isHappy(self, n: int) -> bool:
        def iterNext(n: int) -> int:
            m = 0
            for i in str(n):
                m += int(i) ** 2
            # print(f"{n} -> {m}")
            return int(m)

        a = set()
        while (n := iterNext(n)) != 1:
            if n in a:
                return False
            a.add(n)

        return True


# @lc code=end
sol = Solution()
print(sol.isHappy(19))
print(sol.isHappy(2))
```

```python
# [1299] 将每个元素替换为右侧最大元素
#

# @lc code=start
from typing import List


class Solution:
    def replaceElements(self, arr: List[int]) -> List[int]:
        for i in range(len(arr)):
            arr[i] = max(arr[i + 1 :] + [-1])
        return arr


# @lc code=end

sol = Solution()
print(sol.replaceElements([17, 18, 5, 4, 6, 1]))
print(sol.replaceElements([40]))
```

```python
# @lc code=start
from typing import List


class Solution:
    def lemonadeChange(self, bills: List[int]) -> bool:
        c5, c10 = 0, 0
        for i in bills:
            if i == 5:
                c5 += 1
            elif i == 10:
                if c5 >= 1:
                    c5 -= 1
                else:
                    return False
                c10 += 1
            elif i == 20:
                if c5 >= 1 and c10 >= 1:
                    c5 -= 1
                    c10 -= 1
                elif c5 >= 3:
                    c5 -= 3
                else:
                    return False
        return True


# @lc code=end
sol = Solution()
print(sol.lemonadeChange([5, 5, 5, 10, 20]))
print(sol.lemonadeChange([5, 5, 10, 10, 20]))
print(sol.lemonadeChange([5, 5, 10, 20]))
```
