```python
# [1758] 生成交替二进制字符串的最少操作数
#

# @lc code=start
class Solution:
    def minOperations(self, s: str) -> int:
        a, b, l = 0, 0, ""
        if len(s) % 2 == 1:
            l, s = s[-1], s[:-1]
        for i in range(0, len(s), 2):
            if s[i] != "0":
                a += 1
            if s[i + 1] != "1":
                a += 1
        for i in range(0, len(s), 2):
            if s[i] != "1":
                b += 1
            if s[i + 1] != "0":
                b += 1
        if l != "":
            if l == "1":
                a += 1
            else:
                b += 1
        return min(a, b)


# @lc code=end
```

```python
# [2224] 转化时间需要的最少操作数
#

# @lc code=start
class Solution:
    def convertTime(self, current: str, correct: str) -> int:
        ch, cm = current.split(":")
        oh, om = correct.split(":")
        ch, cm = int(ch), int(cm)
        oh, om = int(oh), int(om)
        c = ch * 60 + cm
        o = oh * 60 + om
        h = (o - c) // 60
        m15 = ((o - c) % 60) // 15
        m5 = (((o - c) % 60) % 15) // 5
        m1 = (((o - c) % 60) % 15) % 5
        return h + m15 + m5 + m1


# @lc code=end
sol = Solution()
print(sol.convertTime(current="02:30", correct="04:35"))
print(sol.convertTime(current="11:00", correct="11:01"))
print(sol.convertTime("09:41", "10:34"))
```

```python
class Solution:
    def largestInteger(self, num: int) -> int:
        s = str(num)
        a = sorted([i for i in s if i in "02468"])
        b = sorted([i for i in s if i in "13579"])
        ans = ""
        for i in s:
            if i in "02468":
                ans += a.pop()
            else:
                ans += b.pop()
        return int(ans)
```

```python
#

# @lc code=start


from itertools import combinations


class Solution:
    def countTriples(self, n: int) -> int:
        return sum(
            2 for i, j, k in combinations(range(1, n + 1), 3) if i * i + j * j == k * k
        )


# @lc code=end

sol = Solution()
print(sol.countTriples(10))
```

```python
# @lc code=start
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        start, end = 1, num
        mid = (start + end) // 2
        while mid * mid != num:
            if mid * mid > num:
                end = mid
                mid = (start + mid) // 2
            else:
                start = mid
                mid = (mid + end) // 2
                for i in range(start, end + 1):
                    if i * i == num:
                        return True
                return False

        return True


# @lc code=end
sol = Solution()
print(sol.isPerfectSquare(100))
```
