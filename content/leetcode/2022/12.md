```python
# [2231] 按奇偶性交换后的最大数字
#

# @lc code=start
class Solution:
    def largestInteger(self, num: int) -> int:
        s = str(num)
        a = sorted([i for i in s if i in "02468"])
        b = sorted([i for i in s if i in "13579"])
        ans = ""
        for i in s:
            if i in "02468":
                ans += a.pop()
            else:
                ans += b.pop()
        return int(ans)


# @lc code=end
sol = Solution()
print(sol.largestInteger(65875))
```

```python
#

# @lc code=start
class Solution:
    def checkZeroOnes(self, s: str) -> bool:
        return max(len(i) for i in s.split("0")) > max(len(i) for i in s.split("1"))
```

```python
# [1925] 统计平方和三元组的数目
#

# @lc code=start


from itertools import combinations


class Solution:
    def countTriples(self, n: int) -> int:
        return sum(
            2 for i, j, k in combinations(range(1, n + 1), 3) if i * i + j * j == k * k
        )


# @lc code=end

sol = Solution()
print(sol.countTriples(10))
```

```python
# [367] 有效的完全平方数
#

# @lc code=start
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        start, end = 1, num
        mid = (start + end) // 2
        while mid * mid != num:
            if mid * mid > num:
                end = mid
                mid = (start + mid) // 2
            else:
                start = mid
                mid = (mid + end) // 2
                for i in range(start, end + 1):
                    if i * i == num:
                        return True
                return False

        return True


# @lc code=end
sol = Solution()
print(sol.isPerfectSquare(100))
```
