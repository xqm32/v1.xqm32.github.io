```python
# [367] 有效的完全平方数
#

# @lc code=start
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        start, end = 1, num
        mid = (start + end) // 2
        while mid * mid != num:
            if mid * mid > num:
                end = mid
                mid = (start + mid) // 2
            else:
                start = mid
                mid = (mid + end) // 2
                for i in range(start, end + 1):
                    if i * i == num:
                        return True
                return False

        return True


# @lc code=end
sol = Solution()
print(sol.isPerfectSquare(100))
```
