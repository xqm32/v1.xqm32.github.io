# [1636] 按照频率将数组升序排序
#

# @lc code=start
from collections import Counter
from functools import cmp_to_key
from itertools import chain
from typing import List


class Solution:
    def frequencySort(self, nums: List[int]) -> List[int]:
        c = Counter(nums)

        def cmp(x, y):
            if c[x] < c[y]:
                return -1
            elif c[x] > c[y]:
                return 1
            else:
                return y - x

        return sorted(nums, key=cmp_to_key(cmp))


# @lc code=end
sol = Solution()
print(sol.frequencySort([2, 3, 1, 3, 2]))

class Solution:
    def findSubarrays(self, nums: List[int]) -> bool:
        s = set()
        for i, j in zip(nums[:-1], nums[1:]):
            if i + j in s:
                return True
            else:
                s.add(i + j)
        return False

#
# [2379] 得到 K 个黑块的最少涂色次数
#

# @lc code=start
class Solution:
    def minimumRecolors(self, blocks: str, k: int) -> int:
        W = blocks[:k].count("W")
        minW = W
        for i in range(0, len(blocks) - k):
            W += (blocks[i + k] == "W") - (blocks[i] == "W")
            minW = min(minW, W)
        return minW


# @lc code=end

sol = Solution()
print(sol.minimumRecolors(blocks="WBBWWBBWBW", k=7))
print(sol.minimumRecolors(blocks="WBWBBBW", k=2))
print(sol.minimumRecolors(blocks="WBB", k=1))

#

# @lc code=start
from typing import List


class Solution:
    def checkXMatrix(self, grid: List[List[int]]) -> bool:
        n = len(grid)
        for i in range(n):
            for j in range(n):
                if i == j or i == (n - j - 1):
                    if grid[i][j] == 0:
                        return False
                elif grid[i][j] != 0:
                    return False
        return True

# @lc code=start
class Solution:
    def strongPasswordCheckerII(self, password: str) -> bool:
        if len(password) < 8:
            return False
        for i, j in zip(password[:-1], password[1:]):
            if i == j:
                return False
        ps: set = set(password)
        if len(set(chr(i) for i in range(97, 123)) & ps) == 0:
            return False
        if len(set(chr(i) for i in range(65, 91)) & ps) == 0:
            return False
        if len(set(chr(i) for i in range(48, 58)) & ps) == 0:
            return False
        if len(set("!@#$%^&*()-+") & ps) == 0:
            return False
        return True


# @lc code=end
sol = Solution()

print(sol.strongPasswordCheckerII("Me+You--IsMyDream"))
print(sol.strongPasswordCheckerII("@Aa1a1a1"))

#

# @lc code=start
class Solution:
    def divisorSubstrings(self, num: int, k: int) -> int:
        a = 0
        s = str(num)
        for i in range(len(s) - k + 1):
            t = int(s[i : i + k])
            a += t != 0 and num % t == 0
        return a


# @lc code=end

#

# @lc code=start
from collections import Counter
from typing import List


class Solution:
    def removeAnagrams(self, words: List[str]) -> List[str]:
        s = [words[0]]
        for i in words[1:]:
            if Counter(i) != Counter(s[-1]):
                s.append(i)
        return s


# @lc code=end
sol = Solution()
print(sol.removeAnagrams(["abba", "baba", "bbaa", "cd", "cd"]))
print(sol.removeAnagrams(["a", "b", "c", "d", "e"]))

# [1758] 生成交替二进制字符串的最少操作数
#

# @lc code=start
class Solution:
    def minOperations(self, s: str) -> int:
        a, b, l = 0, 0, ""
        if len(s) % 2 == 1:
            l, s = s[-1], s[:-1]
        for i in range(0, len(s), 2):
            if s[i] != "0":
                a += 1
            if s[i + 1] != "1":
                a += 1
        for i in range(0, len(s), 2):
            if s[i] != "1":
                b += 1
            if s[i + 1] != "0":
                b += 1
        if l != "":
            if l == "1":
                a += 1
            else:
                b += 1
        return min(a, b)

# [2224] 转化时间需要的最少操作数
#

# @lc code=start
class Solution:
    def convertTime(self, current: str, correct: str) -> int:
        ch, cm = current.split(":")
        oh, om = correct.split(":")
        ch, cm = int(ch), int(cm)
        oh, om = int(oh), int(om)
        c = ch * 60 + cm
        o = oh * 60 + om
        h = (o - c) // 60
        m15 = ((o - c) % 60) // 15
        m5 = (((o - c) % 60) % 15) // 5
        m1 = (((o - c) % 60) % 15) % 5
        return h + m15 + m5 + m1


# @lc code=end
sol = Solution()
print(sol.convertTime(current="02:30", correct="04:35"))
print(sol.convertTime(current="11:00", correct="11:01"))
print(sol.convertTime("09:41", "10:34"))

from itertools import combinations


class Solution:
    def countTriples(self, n: int) -> int:
        return sum(
            2 for i, j, k in combinations(range(1, n + 1), 3) if i * i + j * j == k * k
        )

#

# @lc code=start
class Solution:
    def isPerfectSquare(self, num: int) -> bool:
        start, end = 1, num
        mid = (start + end) // 2
        while mid * mid != num:
            if mid * mid > num:
                end = mid
                mid = (start + mid) // 2
            else:
                start = mid
                mid = (mid + end) // 2
                for i in range(start, end + 1):
                    if i * i == num:
                        return True
                return False

        return True


# @lc code=end
sol = Solution()
print(sol.isPerfectSquare(100))
