```python
# [1614] 括号的最大嵌套深度
#

# @lc code=start
class Solution:
    def maxDepth(self, s: str) -> int:
        ans = 0
        left = 0
        for i in s:
            if i == "(":
                left += 1
                ans = max(ans, left)
            elif i == ")":
                left -= 1
        return ans
# @lc code=end
```
