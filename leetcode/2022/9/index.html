from typing import List


class Solution:
    def specialArray(self, nums: List[int]) -> int:
        nums = sorted(nums, reverse=True)
        for i in range(len(nums) - 1):
            if nums[i] >= i + 1 and nums[i + 1] < i + 1:
                return i + 1
        i = len(nums) - 1
        if nums[i] >= i + 1:
            return i + 1
        return -1


# @lc code=end

sol = Solution()
print(sol.specialArray([3, 5]))
print(sol.specialArray([0, 0]))
print(sol.specialArray([0, 4, 3, 0, 4]))
print(sol.specialArray([3, 6, 7, 7, 0]))
print(sol.specialArray([0]))

from typing import List


class Solution:
    def busyStudent(
        self, startTime: List[int], endTime: List[int], queryTime: int
    ) -> int:
        ans = 0
        for i in range(len(startTime)):
            if endTime[i] >= queryTime >= startTime[i]:
                ans += 1
        return ans

# @lc code=start
class Solution:
    def numberOfMatches(self, n: int) -> int:
        ans = 0
        while n > 1:
            if n % 2 == 0:
                ans += n // 2
                n //= 2
            else:
                ans += (n - 1) // 2
                n = (n - 1) // 2 + 1
        return ans


# @lc code=end
sol = Solution()
print(sol.numberOfMatches(7))
print(sol.numberOfMatches(14))

import re
from typing import List


class Solution:
    def findOcurrences(self, text: str, first: str, second: str) -> List[str]:
        ans = []
        text = text.split()
        for i in range(len(text) - 2):
            if text[i] == first and text[i + 1] == second:
                ans.append(text[i + 2])
        return ans

from collections import Counter
from typing import List


class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        c = Counter(chars)
        ans = 0
        for i in words:
            if c > Counter(i):
                ans += len(i)
        return ans


# @lc code=end

from typing import List


class Solution:
    def findPoisonedDuration(self, timeSeries: List[int], duration: int) -> int:
        ans = 0
        for i in range(0, len(timeSeries) - 1):
            if timeSeries[i + 1] - timeSeries[i] > duration:
                ans += duration
            else:
                ans += timeSeries[i + 1] - timeSeries[i]
        return ans + duration

from typing import List


class Solution:
    def transpose(self, matrix: List[List[int]]) -> List[List[int]]:
        ans = [[0] * len(matrix) for i in matrix[0]]
        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                ans[j][i] = matrix[i][j]
        return ans


# @lc code=end

from typing import List


class Solution:
    def search(self, nums: List[int], target: int) -> int:
        def find(nums: List[int], start: int, end: int, target: int) -> int:
            if start > end:
                return -1
            mid = start + (end - start) // 2
            if nums[mid] > target:
                return find(nums, start, mid - 1, target)
            elif nums[mid] < target:
                return find(nums, mid + 1, end, target)
            else:
                return mid

        return find(nums, 0, len(nums) - 1, target)


# @lc code=end
sol = Solution()
print(sol.search([-1, 0, 3, 5, 9, 12], 9))
print(sol.search(nums=[-1, 0, 3, 5, 9, 12], target=12))

# @lc code=start
class Solution:
    def reverseVowels(self, s: str) -> str:
        r = ""
        self.a = len(s)

        def prev():
            for i in reversed(range(0, self.a)):
                if s[i] in "aeiouAEIOU":
                    self.a = i
                    return i
            return -1

        for i in range(len(s)):
            if s[i] in "aeiouAEIOU":
                r += s[prev()]
            else:
                r += s[i]

        return r


# @lc code=end
sol = Solution()
print(sol.reverseVowels("hello"))
print(sol.reverseVowels("leetcode"))

class Solution:
    def maxDepth(self, s: str) -> int:
        ans = 0
        left = 0
        for i in s:
            if i == "(":
                left += 1
                ans = max(ans, left)
            elif i == ")":
                left -= 1
        return ans
