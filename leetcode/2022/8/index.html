class Solution:
    def makeGood(self, s: str) -> str:
        def nextGood(s: str) -> tuple[str, bool]:
            re = ""
            isGood = True
            i = 0
            while i < len(s) - 1:
                if abs(ord(s[i]) - ord(s[i + 1])) == 32:
                    i += 2
                    isGood = False
                    continue
                re += s[i]
                i += 1
            if i < len(s):
                re += s[i]
            return re, isGood

        isGood = False
        while not isGood:
            s, isGood = nextGood(s)
        return s


# @lc code=end
sol = Solution()
print(sol.makeGood("leEeetcode"))
print(sol.makeGood("abBAcC"))
print(sol.makeGood("s"))
print(sol.makeGood(""))

from typing import List


class Solution:
    def countNegatives(self, grid: List[List[int]]) -> int:
        m = len(grid)
        n = len(grid[0])
        ans = 0

        j = n - 1
        for i in range(m):
            for j in reversed(range(0, j + 1)):
                if grid[i][j] >= 0:
                    ans += n - j - 1
                    print(f"{ans} += {n} - {j+1}")
                    break
            else:
                ans += n
                print(f"{ans} += {n}")

        return ans

from collections import Counter
from typing import List


class Solution:
    def countElements(self, nums: List[int]) -> int:
        a = min(nums)
        b = max(nums)
        if a == b:
            return 0
        d = Counter(nums)
        if d[a] + d[b] == len(nums):
            return 0
        return len(nums) - d[a] - d[b]

from collections import Counter
from functools import reduce
from typing import List


class Solution:
    def hasGroupsSizeX(self, deck: List[int]) -> bool:
        def gcd(a, b):
            return a if b == 0 else gcd(b, a % b)

        deck = Counter(deck).values()
        r = reduce(gcd, deck)
        return r >= 2

from typing import List


class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        if len(prices) < 2:
            return 0

        a = [prices[-1]]
        for i in reversed(prices[:-1]):
            if i > a[-1]:
                a.append(i)
            else:
                a.append(a[-1])
        a = list(reversed(a))

        return max(a[i] - prices[i] for i in range(len(a)))


# @lc code=end

sol = Solution()
print(sol.maxProfit([7, 1, 5, 3, 6, 4]))
print(sol.maxProfit([7, 6, 4, 3, 1]))
print(sol.maxProfit([1]))
print(sol.maxProfit([1, 20]))

class Solution:
    def wordPattern(self, pattern: str, s: str) -> bool:
        di = dict()
        dj = dict()
        s = s.split()
        if len(s) != len(pattern):
            return False
        for i, j in zip(pattern, s):
            if j in dj and dj[j] != i:
                return False
            elif i in di and di[i] != j:
                return False
            else:
                di[i] = j
                dj[j] = i
        return True
# @lc code=end

sol = Solution()
print(sol.wordPattern(pattern = "abba", s = "dog dog dog dog"))
print(sol.wordPattern(pattern = "abba", s = "dog cat cat fish"))
print(sol.wordPattern(pattern = "aaaa", s = "dog cat cat dog"))

class Solution:
    def rearrangeCharacters(self, s: str, target: str) -> int:

            lambda cs: min(

from typing import Optional


class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right


# @lc code=start
# Definition for a binary tree node.


class Solution:
    def sumOfLeftLeaves(self, root: Optional[TreeNode]) -> int:
        def func(root: Optional[TreeNode], isLeft: bool) -> int:
            if root is None:
                return 0
            elif root.left is None and root.right is None and isLeft:
                return root.val
            else:
                return func(root.left, True) + func(root.right, False)

        return func(root, False)

from typing import List


class Solution:
    def isMonotonic(self, nums: List[int]) -> bool:
        for i in range(1, len(nums)):
            if nums[i] < nums[i - 1]:
                nums[i - 1] = -1
            elif nums[i] == nums[i - 1]:
                nums[i - 1] = 0
            elif nums[i] > nums[i - 1]:
                nums[i - 1] = 1
        nums[-1] = 0
        return nums.count(1) == 0 or nums.count(-1) == 0


# @lc code=end
sol = Solution()
print(sol.isMonotonic([1, 2, 2, 3]))
print(sol.isMonotonic([6, 5, 4, 4]))
print(sol.isMonotonic([1, 3, 2]))

from typing import List


class Solution:
    def islandPerimeter(self, grid: List[List[int]]) -> int:
        h = len(grid)
        l = len(grid[0])
        c = 0

        for i in range(h):
            for j in range(l):
                if grid[i][j] == 0:
                    continue
                if i == 0 or grid[i - 1][j] == 0:
                    c += 1
                if i == h - 1 or grid[i + 1][j] == 0:
                    c += 1
                if j == 0 or grid[i][j - 1] == 0:
                    c += 1
                if j == l - 1 or grid[i][j + 1] == 0:
                    c += 1
        return c


# @lc code=end
sol = Solution()
print(sol.islandPerimeter([[0, 1, 0, 0], [1, 1, 1, 0], [0, 1, 0, 0], [1, 1, 0, 0]]))
print(sol.islandPerimeter([[1]]))
print(sol.islandPerimeter([[1, 0]]))

class Solution:
    def isHappy(self, n: int) -> bool:
        def iterNext(n: int) -> int:
            m = 0
            for i in str(n):
                m += int(i) ** 2
            # print(f"{n} -> {m}")
            return int(m)

        a = set()
        while (n := iterNext(n)) != 1:
            if n in a:
                return False
            a.add(n)

        return True

from typing import List


class Solution:
    def lemonadeChange(self, bills: List[int]) -> bool:
        c5, c10 = 0, 0
        for i in bills:
            if i == 5:
                c5 += 1
            elif i == 10:
                if c5 >= 1:
                    c5 -= 1
                else:
                    return False
                c10 += 1
            elif i == 20:
                if c5 >= 1 and c10 >= 1:
                    c5 -= 1
                    c10 -= 1
                elif c5 >= 3:
                    c5 -= 3
                else:
                    return False
        return True
